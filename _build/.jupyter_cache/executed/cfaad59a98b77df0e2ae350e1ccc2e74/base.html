Hs = 5*(s+2)/((s + 3)*(s+10))
\[Hs =\frac{5\,s+10}{{\left(s+3\right)}\,{\left(s+10\right)}}\]
syms t
u_0(t) = heaviside(t);
Xs = laplace(u_0(t))
\[Xs =\frac{1}{s}\]
Ys = Hs*Xs
\[Ys =\frac{5\,s+10}{s\,{\left(s+3\right)}\,{\left(s+10\right)}}\]
Ys = partfrac(Ys)
\[Ys =\frac{5}{21\,{\left(s+3\right)}}-\frac{4}{7\,{\left(s+10\right)}}+\frac{1}{3\,s}\]
yt = ilaplace(Ys)
\[yt =\frac{5\,{\mathrm{e}}^{-3\,t} }{21}-\frac{4\,{\mathrm{e}}^{-10\,t} }{7}+\frac{1}{3}\]
